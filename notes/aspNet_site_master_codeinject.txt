string stdout = "";
string cmd = "whoami";
System.Diagnostics.ProcessStartInfo procStartInfo = new System.Diagnostics.ProcessStartInfo("cmd", "/c " + cmd);

procStartInfo.UseShellExecute = false;
procStartInfo.CreateNoWindow = true;

p.StartInfo = procStartInfo;
